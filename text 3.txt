local StealthEnv = {}

-- Proxy pcall to bypass line number checks
local original_pcall = pcall
local original_xpcall = xpcall

local function sanitize_error(err)
    if type(err) == "string" then
        err = err:gsub(":%d+:", ":1:")
        err = err:gsub("line %d+", "line 1")
    end
    return err
end

_G.pcall = function(func, ...)
    local results = {original_pcall(func, ...)}
    if not results[1] and results[2] then
        results[2] = sanitize_error(results[2])
    end
    return table.unpack(results)
end

_G.xpcall = function(func, errhandler, ...)
    local custom_handler = function(err)
        return errhandler(sanitize_error(err))
    end
    return original_xpcall(func, custom_handler, ...)
end

-- Hide executor signatures
StealthEnv.hidden_globals = {}

function StealthEnv.hide_executor_signatures()
    local signatures = {
        "syn", "secure_load", "fluxus", "getgenv", "getrenv",
        "hookmetamethod", "hookfunction", "getnamecallmethod"
    }
    
    for _, name in ipairs(signatures) do
        if _G[name] then
            StealthEnv.hidden_globals[name] = _G[name]
            _G[name] = nil
        end
    end
end

function StealthEnv.restore_executor_signatures()
    for name, value in pairs(StealthEnv.hidden_globals) do
        _G[name] = value
    end
    StealthEnv.hidden_globals = {}
end

-- Hook debug.getinfo
if debug and debug.getinfo then
    local original_getinfo = debug.getinfo
    
    debug.getinfo = function(level, what)
        local info = original_getinfo(level, what)
        if info then
            if info.source and info.source:find("@") then
                info.source = "@game.script"
            end
            if info.currentline then info.currentline = 1 end
            if info.linedefined then info.linedefined = 1 end
        end
        return info
    end
end

-- Shadow load function
function StealthEnv.execute(script_string)
    StealthEnv.hide_executor_signatures()
    
    local script_func, load_error = loadstring(script_string)
    
    if not script_func then
        return false, "Load Error: " .. tostring(load_error)
    end
    
    local success, exec_error = pcall(script_func)
    
    task.wait(0.3)
    StealthEnv.restore_executor_signatures()
    
    if not success then
        return false, "Execution Error: " .. tostring(exec_error)
    end
    
    return true, "Script executed successfully!"
end

-- Dragging functionality
local dragging, dragInput, dragStart, startPos

TopBar.InputBegan:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
        dragging = true
        dragStart = input.Position
        startPos = MainFrame.Position
        
        input.Changed:Connect(function()
            if input.UserInputState == Enum.UserInputState.End then
                dragging = false
            end
        end)
    end
end)

TopBar.InputChanged:Connect(function(input)
    if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
        dragInput = input
    end
end)

UserInputService.InputChanged:Connect(function(input)
    if input == dragInput and dragging then
        local delta = input.Position - dragStart
        MainFrame.Position = UDim2.new(
            startPos.X.Scale,
            startPos.X.Offset + delta.X,
            startPos.Y.Scale,
            startPos.Y.Offset + delta.Y
        )
    end
end)

-- Button hover effects
local function addHoverEffect(button, normalColor, hoverColor)
    button.MouseEnter:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = hoverColor}):Play()
    end)
    
    button.MouseLeave:Connect(function()
        TweenService:Create(button, TweenInfo.new(0.2), {BackgroundColor3 = normalColor}):Play()
    end)
end
